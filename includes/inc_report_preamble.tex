% --- Bibliography ---

% --- --- Languages --- ---
% Language with polyglossia must be set before biblatex is used, or error ensues.

\usepackage[UKenglish]{babel}
% END --- --- Languages --- ---
\usepackage[ % options to be optimised TODO
  style=verbose
 ,dateabbrev=false
 ,backend=biber
 ,backref=true
 ,style=numeric-comp  % numerical-compressed
 ,sorting=nyt         % name, year, title
 ,sortcites=true      % some other example options ...
 ,block=none
 ,indexing=false
 ,citereset=none
 ,isbn=true
 ,url=true
 ,doi=true            % prints doi
 ,natbib=true         % if you need natbib functions
 ,urldate=long        % shown as: Month daynr, Year
]{biblatex} 
% *** IMPORTANT NOTE ***: Bibliographic entries that contain *any* date, *MUST* set 
% it in AMERICAN FORMAT: YYYY-MM-DD, otherwise biblatex will IGNORE IT! ***
% ---
% Make author surname show up first.
\DeclareNameAlias{sortname}{last-first}
\DeclareNameAlias{default}{last-first}
% Make reference list by bold number followed by dot.
\DeclareFieldFormat{labelnumberwidth}{\mkbibbold{#1.}}
% Chapter name of bibliography section: References
% http://tex.stackexchange.com/questions/56950/trying-to-change-bibliography-references-header
\DefineBibliographyStrings{english}{%
  bibliography = {References},
}
% Date *displayed* format: Month Day, Year.
\DeclareFieldFormat{urldate}{%
  (Accessed: \mkbibmonth{\thefield{urlmonth}}\addspace%
	\thefield{urlday},\addspace%
  \thefield{urlyear}\isdot)}
% --- END Bibliography ---
