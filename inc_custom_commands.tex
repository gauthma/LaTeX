%
% Custom environments
%

% Bullets
% (1) foo
% (2) bar
\newenvironment{enum1}
{
\renewcommand{\theenumi}{\textup{(\arabic{enumi})}}
\renewcommand{\labelenumi}{\theenumi}
\begin{enumerate}}
{\end{enumerate}}

% Bullets
%  i. foo
% ii. bar
\newenvironment{enumi}
{
\renewcommand{\labelenumi}{\theenumi}
\renewcommand{\theenumi}{\textup{\roman{enumi}.}}
\begin{enumerate}}
{\end{enumerate}}

% Bullets
%  I. foo
% II. bar
\newenvironment{enumI}
{
\renewcommand{\labelenumi}{\theenumi}
\renewcommand{\theenumi}{\textup{\Roman{enumi}.}}
\begin{enumerate}}
{\end{enumerate}}
